#include <iostream>

struct Contenedor {
    int puerto,identificacion, peso;
    };
    
int pesototal,indentificacion_total, puertoMP=0, idefMP;
int cont1=0, cont2=0, cont3=0;

int main (void) {
    struct Contenedor contenedores;
    
    for(int i=0; i<5; i++) {
        printf("CONTENEDOR %d\n", i+1);
        
        printf("\nIndentificacion del contenedor (1, 2 o 3):");
        scanf("%d",&contenedores.identificacion);
        
        printf("\nPeso del contenedor (en KG): ");
        scanf("%d",&contenedores.peso);
        
        printf("\nPuerto donde se dejo el contenedor:");
        scanf("%d",&contenedores.puerto);
        
        pesototal += contenedores.peso;
    
        if (contenedores.peso > puertoMP) {
            idefMP = contenedores.identificacion;
        }
        
        if(contenedores.puerto == 1) {
            cont1 += 1;
        }
        if(contenedores.puerto == 2) {
            cont2 += 1;
        }
        if(contenedores.puerto == 3) {
            cont3 += 1;
        }
    }
    
    printf("\nPeso total del buque: %d", pesototal);
    printf("\nEl contenedor con mas peso era de identificacion %d", idefMP);
    printf("\nLa cantidad de contenedor por puerto es:\nPuerto 1: %d\nPuerto 2: %d\nPuerto 3: %d", cont1, cont2, cont3);
}
